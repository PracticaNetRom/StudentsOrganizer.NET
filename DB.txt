CREATE TABLE STUDENTS
(

	id_Student int primary key,
	FirstName varchar(50) not null,
	LastName varchar(30) not null,
	Gender varchar(20) not null,
	BirthDate date,
	Email varchar(30),
	PhoneNumber varchar(10),
	Faculty varchar(50) not null,
	FacultyStartYear date

	
);

CREATE TABLE EVENT
(
	
	id_Event int primary  key,
	NameEvent varchar(40) not null

);


CREATE TABLE EVENT_PE
(
	id_PE primary key,
	idStudent int,
	idEvent int,
	Period date,
	Task varchar(200),
	Remarsk varchar(200)

);
